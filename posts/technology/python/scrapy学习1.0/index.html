import scrapy
from scrapy.linkextractors import LinkExtractor
from scrapy.spiders import CrawlSpider, Rule
from ..items import ScrapyReadbookItem

class ReadSpider(CrawlSpider):
    name = 'read'
    allowed_domains = ['www.dushu.com']
    start_urls = ['https://www.dushu.com/book/1158_1.html']

    rules = (
        Rule(LinkExtractor(allow=r'/book/1158_\d+.html'), callback='parse_item', follow=False),
    )

    def parse_item(self, response):
        name = response.xpath('//div[@class="book-info"]/h3/a/@title').getall()
        book = ScrapyReadbookItem(name=name)
        yield book
