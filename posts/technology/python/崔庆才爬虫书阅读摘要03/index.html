MySQL(关系型数据库)#

MySQL的学习将以代码的形式来展开，并辅以少量文字解释。

sql语句使用格式化符%s来构造，避免使用+的拼接字符串、对于数据的插入、更新、删除操作都需要db的commit方法才行，如果执行失败则使用rollback执行回调。标准写法如下：

1
2
3
4
5

try:
    cursor.execute(sql, (带传入的值))
    db.commit()
except:
    db.rollback()

import pymysql

person1 = {"id":"2022812", "name":"Jhon", "age":18}

# 声明了一个MySQL连接对象 db
db = pymysql.connect(host="localhost",  # MySQL运行的host（即IP）
                     user="root",   # 用户名
                     password="1234",   # 密码
                     database="spiders")    # 所连接的数据库

# 获得MySQL的操作游标，可以利用游标执行sql语句
cursor = db.cursor()
sql = "insert into students(id, name, age) values(%s, %s, %s)"
try:
    cursor.execute(sql, (person1["id"], person1["name"], person1["age"]))
    db.commit()
except:
    db.rollback()

db.close()

import pymysql

data = {"id":"202281203", "name":"Saliy", "age":18}
table = "students"
keys = ", ".join(data.keys())	# output："id", "name", "age"
values = ", ".join(['%s'] * len(data))
# 声明了一个MySQL连接对象 db
db = pymysql.connect(host="localhost",  # MySQL运行的host（即IP）
                     user="root",   # 用户名
                     password="1234",   # 密码
                     database="spiders")    # 所连接的数据库

# 获得MySQL的操作游标，可以利用游标执行sql语句
cursor = db.cursor()
sql = f"insert into {table}({keys}) values({values})"
try:
    cursor.execute(sql, tuple(data.values()))
    print("Successful")
    db.commit()
except:
    print("Failed")
    db.rollback()

db.close()

实现了动态SQL语句的实现，sql中的列名就是通过传入数据字典的键名来表示，excute方法的第一个参数是sql变量，第二个参数是传入data的值所构造的元组实现。

import pymysql

data = {"id":"202281204", "name":"tom", "age":118}
table = "students"
keys = ", ".join(data.keys())
values = ", ".join(['%s'] * len(data))
# print(values)
# 声明了一个MySQL连接对象 db
db = pymysql.connect(host="localhost",  # MySQL运行的host（即IP）
                     user="root",   # 用户名
                     password="1234",   # 密码
                     database="spiders")    # 所连接的数据库

# 获得MySQL的操作游标，可以利用游标执行sql语句
cursor = db.cursor()
# ON DUPLICATE KEY UPDATE是如果主键已经存在就会执行更新，否则就是插入
sql = f"insert into {table}({keys}) values ({values}) ON DUPLICATE KEY UPDATE "
update = ",".join(["{key} = %s".format(key=key) for key in data])

sql += update
try:
    if cursor.execute(sql, tuple(data.values())*2):
        print("Successful")
        db.commit()
except:
    print("Failed")
    db.rollback()

db.close()

import pymysql

table = "students"
condition = "age < 100"
# 声明了一个MySQL连接对象 db
db = pymysql.connect(host="localhost",  # MySQL运行的host（即IP）
                     user="root",   # 用户名
                     password="1234",   # 密码
                     database="spiders")    # 所连接的数据库

# 获得MySQL的操作游标，可以利用游标执行sql语句
cursor = db.cursor()

try:
    cursor.execute(sql)
    print("查询的数据条数为：", cursor.rowcount)
    one = cursor.fetchone()
    print("第一条：", one)
    alldata = cursor.fetchall()
    print(alldata)
    print("--------")
    for i in alldata:
        print(i)
except:
    print("ERROR")
db.close()

符号 含义 例子
$lt 小于（less than） result = collection.find({“age”:{"$lt":50}})
$gt 大于（greater than） result = collection.find({“age”:{"$gt":50}})
$lte 小于等于 同上
$gte 大于等于 同上
$ne 不等于（not equal to） 同上
$in 在范围内 result = collection.find({“age”:{"$in":[18, 23]}})
$nin 不在范围内 同上
$regex 匹配正则表达式 语法同上
$exists 属性是否存在 同上
$text 文本查询 {"$text":{"$search":“hanson”}}

import pymongo
from bson.objectid import ObjectId
client = pymongo.MongoClient(host="localhost")

db = client["test"]
collection = db["students"]
condition = {"name":"tom"}
# 查询要更新的数据
student = collection.find_one(condition)
# 修改数据
student["age"] = 100
# 调用update_one方法将 原条件 和 修改后的数据传入
result = collection.update_one(condition, {"$set":student}, upsert=True)
print(result)

更新后的结果可以调用matched_count和modified_count属性获得匹配的数据条数和影响的数据条数

并且更新推荐使用{"$set":student}的更新方式，因为这样可以只更新student字典内存在的字段，如果原先还有其他字段，则既不会更新，也不会删除。而如果不使用$set，就会把之前的数据全部使用student字典替换，要是原本存在其他字段，则会被删除。
