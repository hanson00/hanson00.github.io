import execjs
import requests
import time
import pymongo

with open("qimai.js", "r", encoding="utf8") as f:
    data = f.read()
context = execjs.compile(data)
analysis = context.call('run')

now_date = str(time.strftime("%Y-%m-%d", time.localtime()))
now_hms = time.strftime("%H:%M:%S", time.localtime())

class Qimai:
    def __init__(self):
        self.db = pymongo.MongoClient(host="localhost")
        self.collection = self.db['spider']['qimai']
        self.url = 'https://api.qimai.cn/rank/index'
        self.analysis = analysis
        self.headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/107.0.0.0 Safari/537.36"
        }

    def save_date(self, item):
        res = self.collection.insert_one(item)
        print(res)

    def get_data(self):
        for page in range(1, 3):
            with open("qimai.js", "r", encoding="utf8") as f:
                data = f.read()
            context = execjs.compile(data)
            analysis = context.call('run', page, now_date, now_hms)
            params = {
                "analysis": analysis,
                "brand": "free",  # 爬取其他改该参数即可
                "device": "ipad",  # 爬取其他改该参数即可
                "country": "cn",  # 爬取其他改该参数即可
                "genre": "36",
                "date": now_date,
                "page": page,
                "is_rank_index": "1",
                "snapshot": now_hms
            }
            response = requests.get(url=self.url, params=params, headers=self.headers)
            data_lists = response.json()['rankInfo']
            for data in data_lists:
                item = {}
                item['app_index'] = data['index']
                item['app_name'] = data['appInfo']['appName']
                item['app_price'] = data['appInfo']['price']
                item['app_comment_num'] = data['comment']['num']
                item['app_comment_rating'] = data['comment']['rating']
                self.save_date(item)

    def run(self):
        self.get_data()

if __name__ == '__main__':
    qimai = Qimai()
    qimai.run()
