def get_baidu_hot():
    url = "https://top.baidu.com/board?tab=realtime"
    response = requests.get(url=url)
    # print(response.text)
    reword = re.compile('"word":"(?P<content>.*?)"')
    result = reword.finditer(response.text)
    # for i in result:
    #     print(i.group("content"))
    content = [i.group("content") for i in result]
    # print(content)
    return content
    response.close()
