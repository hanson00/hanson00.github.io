import win32com.client
from win32com.client import Dispatch
xlapp = win32com.client.Dispatch("Excel.Application")       #创建一个Excel程序
xlapp.DisplayAlerts = False     #取消弹窗
xlapp.Visible = True        #显示Excel
wb = xlapp.Workbooks.Add()      #添加工作簿
wb.Worksheets.Item("Sheet1").Range("A1").Value = 2000       #为单元格赋值
wb.Worksheets("Sheet1").Range("A2").Value = 100     #为单元格赋值
wb.Worksheets("Sheet1").cells(1,3).value = 200      #为单元格赋值
wb.SaveAs(r"C:\Users\Lenovo\Desktop\jb\123.xlsx")       #保存工作簿
wb.Close()      #关闭工作簿
xlapp.quit()        #关闭Excel程序
print("正常结束")

from win32com.client import Dispatch
# 创建Excel程序
xlapp = Dispatch("Excel.Application")
# 视为可见
xlapp.Visible = True
# 警告信息
xlapp.DisplayAlerts = True
# 添加工作簿
wb = xlapp.Workbooks.Add()
ws = wb.Sheets("Sheet1")		#或者ws = wb.Sheets.Add()后面这个为新添加一个Sheet
