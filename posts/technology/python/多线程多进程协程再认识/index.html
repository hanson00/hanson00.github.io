import ts
import time
from threading import Thread
def single_thread():
    print("single_thread being...")
    for url in ts.urls:
        ts.craw(url)
    print("single_thread ending...")

def multi_thread():
    print("single_thread being...")
    thread_list = []
    for url in ts.urls:
        thread_list.append(Thread(target=ts.craw, args=(url,)))

    for thread in thread_list:
        thread.start()
    for thread in thread_list:
        thread.join()
    print("single_thread ending...")

from urllib.parse import urljoin

urls = [f"https://www.cnblogs.com/#p{page}" for page in range(1, 50)]
# print(urls)

def craw(url):
    response = requests.get(url=url)
    return response.text

def parse(data):
    html = etree.HTML(data)
    li_s = html.xpath("//div[@class='post-item-text']")
    res_li = []
    for li in li_s:
        title = li.xpath("./a/text()")[0]
        href = li.xpath("./a/@href")[0]
        title_href = urljoin(old_url, href)
        # print(title_href, title)
        res_li.append([title_href, title])
    return res_li

if __name__ == '__main__':
    a = parse(craw(urls[0]))
    print(a)
