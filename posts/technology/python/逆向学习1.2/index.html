定义了一个变量 opening 保留原始 XMLHttpRequest.open 方法，然后重写 XMLHttpRequest.open 方法，判断如果 rnd 字符串值在 URL 里首次出现的位置不为 -1，即 URL 里包含 analysis字符串，则执行 debugger 语句，会立即断下。

(function(){
    // 赋值后重写该方法
    var opening = window.XMLHttpRequest.prototype.open
    window.XMLHttpRequest.prototype.open = function(method, url, async){
        if (url.indexOf('analysis') != -1){
            debugger
        }
        return opening.apply(this, arguments)
    }
})()

**apply：**调用一个对象的一个方法，用另一个对象替换当前对象。例如：B.apply(A, arguments);即A对象应用B对象的方法。即以当前的对象去调用B方法上面就是opening = window.XMLHttpRequest.prototype.open

Interceptors-拦截器#

前端给服务器发送请求，后端给客户端发送响应，一般vue都会写

  • 请求拦截器：在发送请求之前，可以借助一些函数来对请求的内容和参数做一些检测。若有问题可以直接取消请求。

  • 响应拦截器：当服务器返回响应数据时，响应拦截器会在我们拿到结果前预先处理响应数据。例如对响应数据做一些格式化处理，或者当响应失败时，可以做一些失败提醒和纪录。

axios = require('axios')
//设置请求拦截器
axios.interceptors.request.use(function (config) {
    console.log('请求拦截器 成功')
    config.timeout = 2000; //修改请求config
    config.headers['sign'] = 'lili'
    return config;
}, function (error) {
    console.log('请求拦截器 失败')
    return Promise.reject(error);
});

//设置响应拦截器
axios.interceptors.response.use(function (response) {
    console.log('响应拦截器 成功')
    console.log('调解密函数进行解密数据')
    //return response;
    return response.data; //修改响应数据
}, function (error) {
    console.log('响应拦截器 失败')
    return Promise.reject(error);
});

//发送请求
axios.get('http://httpbin.org/get').then(res=>console.log(res))

(function () {
  var cookieTemp = '';
  Object.defineProperty(document, 'cookie', {
    set: function (val) {
      if (val.indexOf('v') != -1) {
        debugger;
      }
      console.log('Hook捕获到cookie设置->', val);
      cookieTemp = val;
      return val;
    },
    get: function () {
      return cookieTemp;
    },
  });
})();

const express = require('express')
const app = express()
const port = 5001

// 模拟加密方法
function jiami(val){
    var res = btoa(val)
    return res
}

app.get('/', (req, res) => {
    res.send('helloo')
})

app.get('/passparam', function (req, res) {
    // 获取URL的请求传参
    var params = req.query
    // res.send(params)
    res.send(params.name)
})

app.get('/jiami', function (req, res) {
    var params = req.query
    var jiamires = jiami(params.password)
    res.send(jiamires)
})

app.listen(port, () => {
    console.log(`Example app listenging on ${port}`)
})
