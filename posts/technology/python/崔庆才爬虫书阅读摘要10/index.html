import requests
from urllib.parse import urljoin

BASE_URL = "https://login3.scrape.center/"
LOGIN_URL = urljoin(BASE_URL, "/api/login")
INDEX_URL = urljoin(BASE_URL, "/api/book")
USERNAME = "admin"
PASSWORD = "admin"
json = {
'username' : USERNAME,
'password' : PASSWORD
}
response_login = requests.post(url=LOGIN_URL, json=json)

data = response_login.json()
print("data JSON is ", data)

jwt = data.get('token')
print("JWT is ", jwt)

headers = {
    'Authorization': f"jwt {jwt}"
}

response_index = requests.get(url=INDEX_URL, params={'limit':18, 'offset':0}, headers=headers)

print(response_index.status_code)
print(response_index.url)
print(response_index.json())
