import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
url = "https://www.lagou.com/"
browser = webdriver.Chrome()
browser.get(url)
time.sleep(1)
browser.find_element(By.XPATH,'//*[@id="changeCityBox"]/p[1]/a').click()
time.sleep(1)
browser.find_element(By.XPATH,'//*[@id="search_input"]').send_keys("爬虫", Keys.ENTER)
time.sleep(1)

#接下来提取公司名称，岗位名称，工作地点
# 查找存放数据的位置，进行数据提取
#进行列表全部数据的粗提取
all_list = browser.find_elements(By.XPATH, value='//*[@id="jobList"]/div[1]/div')

for li in all_list:
    li.find_element(By.XPATH, value='./div[1]/div[1]/div[1]/a').click()
#                                             //*[@id="jobList"]/div[1]/div[1]/div[1]/div[1]/div[1]/a
    time.sleep(1)
    browser.switch_to.window(browser.window_handles[-1])    #切换窗口，将它置为最后一个窗口
    time.sleep(1)
    datajob = browser.find_element(By.XPATH, value='//*[@id="job_detail"]/dd[2]').text
    print(datajob)
    browser.close()
    browser.switch_to.window(browser.window_handles[0])
    time.sleep(1)

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
import time
from selenium.webdriver.common.action_chains import  ActionChains #鼠标操作导包,事件链
#防止被认出是爬虫程序，chrome版本88以上适用
chrome_opt = webdriver.ChromeOptions()
chrome_opt.add_argument('--disable-blink-features=AutomationControlled')
browser = webdriver.Chrome(options=chrome_opt)

url = 'https://kyfw.12306.cn/otn/resources/login.html'
browser.get(url)
time.sleep(1)
#输入账号密码
browser.find_element(By.XPATH, value='//*[@id="J-userName"]').send_keys('你的账号')
browser.find_element(By.XPATH, value='//*[@id="J-password"]').send_keys('密码')
browser.find_element(By.XPATH, value='//*[@id="J-login"]').click()
time.sleep(2)

#拖拽目标按钮
btn = browser.find_element(By.XPATH, value='//*[@id="nc_1_n1z"]')

#找到目标拖拽验证码
browser.find_element(By.XPATH, value='//*[@id="nc_1__scale_text"]/span')


#拖拽移动
ActionChains(browser).drag_and_drop_by_offset(btn, 300, 0).perform()  #横推拽

Selenium的等待#

Selenium的等待有以下三种：

  1. 强制等待time.sleep()等到网页加载完成再等待指定的秒数才能继续往下执行。
  2. 隐式等待，在指定等待时间到来前加载完了，直接继续往下执行，如果等待了指定的时间后还未成功打开网页就直接继续往下执行。
  3. 显示等待，最为灵活。两个人去玩，其中一人到了目的地，另一人告诉他不来了，他就一个人进去。

隐式等待判断网页是否加载完，但可能网页上某个JavaScript脚本加载速度特别慢，而JavaScript脚本只是个限制脚本，不影响网页的正常浏览，但仍然需要等待全部页面加载完成。

显示等待只需要判断某个特定元素是否加载出来就可以了。

隐式等待如下：隐式等待设置一次就会在整个程序中都起作用

chrome.quit()

from selenium import webdriver
from selenium.webdriver.support import expected_conditions
from selenium.webdriver.support.wait import WebDriverWait
from selenium.webdriver.common.by import By

url = 'https://www.csdn.net/'

chrome = webdriver.Chrome()
# 设置隐式等待，等待时间为隐式等待和显示等待时间长的那个
chrome.implicitly_wait(10)
chrome.get(url=url)

try:
    WebDriverWait(chrome, 20, 0.5).until(expected_conditions.presence_of_element_located((By.LINK_TEXT, u'博客')))
finally:
    print(chrome.find_element(By.LINK_TEXT, "博客").get_attribute('href'))
    chrome.quit()

WebDriverWait()可选择的参数有四个，一般情况下只用前三个，第一个参数是设置要打开的浏览器，第二个参数是设置超时时间，第三个是设置检查频率。

后面一部分的until(method,message="")，message返回的结果是0，即失败后要返回的消息。

向expected_conditions.presence_of_element_located((元组))
