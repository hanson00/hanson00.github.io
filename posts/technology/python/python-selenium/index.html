from selenium import webdriver
#使用selenium获取网页的基本操作
url = "http://finance.sina.com.cn/realstock/company/sh000001/nc.shtml"
browser = webdriver.Chrome()  #声明模拟的浏览器是Chrome
browser.maximize_window()  #最大化浏览器
browser.get(url) #模拟浏览器发送请求

browser.quit() #关闭模拟浏览器

#使用无界面浏览器是为了在某些爬取工作时不弹出浏览器窗口来影响自己的操作

from selenium import webdriver

url = "http://finance.sina.com.cn/realstock/company/sh000001/nc.shtml"

chrome_opt = webdriver.ChromeOptions()
chrome_opt.add_argument("--headless")
browser = webdriver.Chrome(options=chrome_opt)  #声明模拟的浏览器是Chrome
browser.get(url)  #控制浏览器发起请求，访问网址

import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
url = "https://www.lagou.com/"
browser = webdriver.Chrome()
browser.get(url)
time.sleep(1)
browser.find_element(By.XPATH,'//*[@id="changeCityBox"]/p[1]/a').click()
time.sleep(1)
browser.find_element(By.XPATH,'//*[@id="search_input"]').send_keys("爬虫", Keys.ENTER)
time.sleep(1)

#接下来提取公司名称，岗位名称，工作地点
# 查找存放数据的位置，进行数据提取
#进行列表全部数据的粗提取
all_list = browser.find_elements(By.XPATH, value='//*[@id="jobList"]/div[1]/div')

for li in all_list:
    li.find_element(By.XPATH, value='./div[1]/div[1]/div[1]/a').click()
#                                             //*[@id="jobList"]/div[1]/div[1]/div[1]/div[1]/div[1]/a
    time.sleep(1)
    browser.switch_to.window(browser.window_handles[-1])    #切换窗口，将它置为最后一个窗口
    time.sleep(1)
    datajob = browser.find_element(By.XPATH, value='//*[@id="job_detail"]/dd[2]').text
    print(datajob)
    browser.close()
    browser.switch_to.window(browser.window_handles[0])
    time.sleep(1)
