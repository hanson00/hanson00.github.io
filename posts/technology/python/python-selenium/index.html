from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
import time, random
import pymongo

class Yw:
    def __init__(self):
        chrome_opt = webdriver.ChromeOptions()
        chrome_opt.add_argument("--disable-blink-features=AutomationControlled")
        self.chrome = webdriver.Chrome(options=chrome_opt)
        self.url = "https://www.yiwugo.com/"
        self.db = pymongo.MongoClient(host="localhost")
        self.collection = self.db["test"]["yiwu"]

    def visit_url(self):
        self.chrome.get(url=self.url)
        self.chrome.find_element(By.NAME, value='q').send_keys("ææ€ª", Keys.ENTER)
        self.son_page()

    def drop_down(self):
        for i in range(1, 10):
            time.sleep(random.randint(100, 200)/ 1000)
            js1 = f"window.scrollTo(0, document.body.scrollHeight* {i / 10})"
            self.chrome.execute_script(js1)

    def son_page(self):
        self.drop_down()
        targets = self.chrome.find_elements(By.XPATH, value='//*[@id="newframe"]/div/div/div[2]/div[3]/div')
        for target in targets:
            item = {}
            target_name = target.find_element(By.XPATH, value="./ul/li[1]/a").get_attribute('title')
            target_price = target.find_element(By.XPATH, value="./ul/li[2]/span[1]/em").text
            target_address = target.find_element(By.XPATH, value="./ul/li[4]").text
            item["name"] = target_name
            item["price"] = target_price
            item["address"] = target_address
            self.save_data(item)
        self.next_page()

    def next_page(self):
        next_btn = self.chrome.find_element(By.XPATH, value="//a[@class='page_next_yes']")
        try:
            if next_btn:
                next_btn.click()
                time.sleep(0.1)
                self.son_page()
            else:
                self.chrome.quit()
        except Exception as e:
            print(e)

    def save_data(self, data):

        res = self.collection.insert_one(data)
        print("ä¿å­˜æˆåŠŸï¼š",res)

    def run(self):
        self.visit_url()

    yiwu = Yw()
    yiwu.run()
