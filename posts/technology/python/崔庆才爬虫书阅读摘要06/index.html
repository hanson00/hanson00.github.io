import asyncio
import time
import aiohttp

async def get_resp_text(url):
    async with aiohttp.ClientSession() as session:
        response = await session.get(url=url)
        await response.text()
        return response

async def request():
    url = "https://www.httpbin.org/delay/5"
    print(f"wait for {url}")
    response = await get_resp_text(url)
    print(f"GET from {response}")

task = [asyncio.ensure_future(request()) for _ in range(3)]

loop.run_until_complete(asyncio.wait(task))

end_time = time.time()
print(f"spending time is {end_time - start_time}")
