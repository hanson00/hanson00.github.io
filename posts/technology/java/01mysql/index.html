MySQL的学习#

1.命令行的基本命令#

数据库操作：通过命令行登录MySQL使用命令:mysql -uroot -p1234 ( -uMySQL用户名,-pMySQL密码 )

show databases; 查询本地数据库

create database 数据库名称; 创建数据库

一般采用该语句用于创建数据库

create database if not exists 数据库名称;

drop database 数据库名称; 删除数据库

一般采用该语句用于删除数据库

drop database if exists 数据库名称;

use 数据库名称; 使用数据库

select database(); 查看当前使用的数据库

表操作：

show tables; 查看当前数据库的表名称

desc 表名称; 查询表结构

--创建表
create table 表名(

列名1,数据类型，

列名2,数据类型，

列名3,数据类型，

列名n,数据类型)；

drop table 表名; 删除表

drop table if exists 表名; 删除表并判断是否存在

alter table 表名 rename 新表名 修改表名

alter table 表名 add 新列名 数据类型; 添加一列

alter table 表名 modify 列名 新的数据类型; 修改数据类型

alter table 表名 change 列名 新的数据类型; 修改列名和数据类型

alter table 表名 drop 列名; 删除列

insert into 表名(列名1,列名2,...) values(值1,值2,值3); 给指定的列添加数据

insert into 表名 values(值1,值2,值3,...); 给全部列添加数据

insert into 表名 values(值1,值2,值3,...),(值1,值2,值3,...),(值1,值2,值3,...); 批量添加数据
