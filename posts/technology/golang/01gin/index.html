package main

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

type Person struct {
	age  int
	name string
}

func main() {
	//创建一个默认路由
	route := gin.Default()

	//配置模板文件
	route.LoadHTMLGlob("templates/*")

	//配置路由
	route.GET("/", func(c *gin.Context) {
		c.String(http.StatusOK, "欢迎来到首页")
	})

	route.POST("/post", func(c *gin.Context) {
		c.String(http.StatusOK, "欢迎来到post页面")
	})

	route.GET("json1", func(c *gin.Context) {
		c.JSON(http.StatusOK, map[string]interface{}{
			"json1":     "666",
			"第一个json页面": "666",
		})
	})

	route.GET("json2", func(c *gin.Context) {
		c.JSON(http.StatusOK, gin.H{
			"json2":     "666",
			"第2个json页面": "666",
		})
	})

	route.GET("json3", func(c *gin.Context) {
		tom := &Person{
			18,
			"tom",
		}
		c.JSON(http.StatusOK, tom)
	})

	route.GET("xml", func(c *gin.Context) {
		c.XML(http.StatusOK, map[string]interface{}{
			"xml": "6666",
		})
	})

	route.GET("index", func(c *gin.Context) {
		c.HTML(http.StatusOK, "index.html", map[string]interface{}{
			"title": "后台的标题",
		})
	})

	route.Run(":8821")
}

package main

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

type Aticle struct {
	Title   string
	Content string
}

func main() {
	//创建一个默认路由
	route := gin.Default()

	//配置模板路径
	//route.LoadHTMLGlob("templates/*")
	//如果有多层模板目录/**代表一层目录
	route.LoadHTMLGlob("templates/**/*")

	//前台路由
	route.GET("/", func(c *gin.Context) {
		//这里的html名称要和渲染的htmldefine后面的名称一致
		c.HTML(http.StatusOK, "templates/default/beindex.html", gin.H{
			"title": "首页",
		})
	})

	route.GET("/new", func(c *gin.Context) {
		news := &Aticle{
			"新闻标题",
			"内容在这里",
		}
		c.HTML(http.StatusOK, "templates/default/benew.html", gin.H{
			"title": "首页",
			"new":   news,
		})
	})

	//后台路由
	route.GET("/admin", func(c *gin.Context) {
		c.HTML(http.StatusOK, "templates/admin/adindex.html", gin.H{
			"title": "首页",
		})
	})

	route.GET("/adminnew", func(c *gin.Context) {
		news := &Aticle{
			"新闻标题",
			"内容在这里",
		}
		c.HTML(http.StatusOK, "templates/admin/adnew.html", gin.H{
			"title": "首页",
			"new":   news,
		})
	})
	route.Run(":8848")
}
