单节点替换函数#

path.replaceWith(types.valueToNode(value)); 这行代码用于将 AST 中的节点替换为新的节点。

path.replaceWith() 方法被调用，并传入 types.valueToNode(value)，这将创建一个新的 AST 节点，其值为变量 value 中的结果。然后，这个新节点将替换原始的二元表达式节点。

types.valueToNode(value) 是 Babel 提供的一个方法，用于根据给定的值创建相应的 AST 节点。在这种情况下，它会创建一个表示 value 的数字字面量节点。

这样，通过 path.replaceWith(types.valueToNode(value))，你可以将原始的二元表达式节点替换为一个新的节点，从而实现对抽象语法树的修改。

const add = {
    BinaryExpression(path){
        let {left, right, operator} = path.node
        if (types.isNumericLiteral(left) && types.isNumericLiteral(right) && operator === '+')
        {
            value = left.value + right.value
            path.replaceWith(types.valueToNode(value))
        }
        if (types.isStringLiteral(left) && types.isStringLiteral(right) && operator === '+')
        {
                        value = left.value + right.value
            path.replaceWith(types.valueToNode(value))
        }
    }
}
traverse(ast, add)

returnArray = {
    CallExpression(path){
        let {callee,arguments} = path.node
        let object = callee.object.value;
        let property = callee.property.value;
        let argument = arguments[0].value;
         array = object[property](argument)
         path.replaceWithMultiple(types.valueToNode(array))
    }
}
traverse(ast,returnArray)

const constantFold = {
    "BinaryExpression|UnaryExpression|ConditionalExpression"(path) {
        // 防止溢出
        if(path.isUnaryExpression({operator:"-"}) ||
    	   path.isUnaryExpression({operator:"void"}))
    	{
    		return;
    	}
        const {confident, value} = path.evaluate();
        if (!confident)
            return;
        if (typeof value == 'number' && (!Number.isFinite(value))) {
            return;
        }
        path.replaceWith(types.valueToNode(value));
    },
}
traverse(_ast, constantFold);

const transform_literal = {
  NumericLiteral({node}) {
    if (node.extra && /^0[obx]/i.test(node.extra.raw)) {
      node.extra = undefined;
    }
  },
  StringLiteral({node})
  {
    if (node.extra && /\\[ux]/gi.test(node.extra.raw)) {
      node.extra = undefined;
    }
  },
}
traverse(ast,transform_literal)
