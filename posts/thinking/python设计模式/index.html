import threading

class Singleton:
    instance = None
    Lock = threading.RLock()
    def __init__(self, name):
        self.name = name

    def __new__(cls, *args, **kwargs):
        if cls.instance:
            return cls.instance
        # 在锁上面加入的两条语句可以稍微减少点系统资源的使用
        with cls.Lock:
            if cls.instance:
                return cls.instance
            cls.instance = object.__new__(cls)
            return cls.instance

def task():
    obj = Singleton("tom")
    print(obj)

for i in range(10):
    t = threading.Thread(target=task)
    t.start()

目的：工厂模式实现了创建者和调用者的分离，使用专门的工厂类选择实现类、创建对象进行统一的管理和控制。

class CarFactory:

    def say_brand(self):
        print("告诉你我是什么车")

    def create_car(self, name):
        if name == "奔驰":
            return BenZ()
        elif name == "比亚迪":
            return BYD()
        elif name == "特斯拉":
            return TSLA()
        else:
            return "该车不在该工厂生产"

class BenZ(CarFactory):

    def say_brand(self):
        print("我是梅赛德斯奔驰")

class BYD(CarFactory):

    def say_brand(self):
        print("我是比亚迪")

class TSLA(CarFactory):

    def say_brand(self):
        print("我是特斯拉")

# 创建工厂类，用于统一创建对象
factory = CarFactory()
car1 = factory.create_car("比亚迪")
car2 = factory.create_car("特斯拉")
print(car2)
print(car1)
car2.say_brand()
